\begin{itemize}
  \begin{paracol}{2}
    \item[] Veni, Creátor Spíritus, \\
    mentes tuórum vísita, \\
    imple supérna grátia, \\
    qu{\ae} tu creásti péctora.

    \switchcolumn

    \item[] \emph{Oh vinde, Espírito Criador, \\
    as nossas almas visitai \\
    e enchei os nossos corações \\
    com vossos dons celestiais.}

    \switchcolumn*

    \item[] Qui díceris Paráclitus, \\
    donum Dei altíssimi, \\
    fons vivus, ignis, cáritas, \\
    et spiritális únctio.

    \switchcolumn

    \item[] \emph{Vós sois chamado o Intercessor, \\
    do Deus excelso o dom sem par, \\
    a fonte viva, o fogo, o amor, \\
    a unção divina e salutar.}

    \switchcolumn*

    \item[] \emph{Tu septifórmis múnere, \\
    dextr{\ae} Dei tu dígitus, \\
    tu rite promíssum Patris, \\
    sermóne ditans gúttura.}

    \switchcolumn

    \item[] \emph{Sois doador dos sete dons, \\
    e sois poder na mão do Pai, \\
    por ele prometido a nós, \\
    por nós seus feitos proclamais.}

    \switchcolumn*

    \item[] Accénde lumen sénsibus; \\
    infúnde amórem córdibus, \\
    infírma nostri córporis \\
    virtúte firmans pérpeti.

    \switchcolumn

    \item[] \emph{A nossa mente iluminai, \\
    os corações enchei de amor, \\
    nossa fraqueza encorajai, \\
    qual força eterna e protetor.}

    \switchcolumn*

    \item[] Hostem repéllas lóngius, \\
    pacémque dones prótinus; \\
    ductóre sic te pr\'{\ae}vio \\
    vitémus omne nóxium.

    \switchcolumn

    \item[] \emph{Nosso inimigo repeli, \\
    e concedei-nos vossa paz; \\
    se pela graça nos guiais, \\
    o mal deixamos para trás.}

    \switchcolumn*

    \item[] Per te sciámus da Patrem, \\
    noscámus atque Fílium; \\
    te utriúsque Spíritum \\
    credámus omni témpore.

    \switchcolumn

    \item[] \emph{Ao Pai e ao Filho Salvador \\
    por vós possamos conhecer. \\
    Que procedeis do seu amor \\
    fazei-nos sempre firmes crer.}

    \switchcolumn*

    \item[] Amen

    \switchcolumn

    \item[] \emph{Amém.}
  \end{paracol}
\end{itemize}
